    //link transmit
    int8_t link_transmit(int16_t transmit_delay,int16_t total,linkID_t Link_id){
      /*start*/
      int16_t count = 0;
      for(count=0;count<total;count++){
        uint8_t  msg[10];
        msg[1]=count;//low bit
        msg[2]=count>>8;//high bit
      
        SMPL_Send(Link_id, msg, sizeof(msg));

        MRFI_DelayMs(transmit_delay);
      }
      return 1;
    }

    smplStatus_t setPowerMode(uint8_t level)
    {
      //ioctlLevel_t PowerLevel;
      uint8_t PowerLevel;
      //PowerLevel=level;
      switch(level){
        case 0:
          PowerLevel=IOCTL_LEVEL_0;
          return SMPL_Ioctl( IOCTL_OBJ_RADIO, IOCTL_ACT_RADIO_SETPWR, &PowerLevel);          
        case 1:
          PowerLevel=IOCTL_LEVEL_1;
          return SMPL_Ioctl( IOCTL_OBJ_RADIO, IOCTL_ACT_RADIO_SETPWR, &PowerLevel);                   
        case 2:
          PowerLevel=IOCTL_LEVEL_2;
          return SMPL_Ioctl( IOCTL_OBJ_RADIO, IOCTL_ACT_RADIO_SETPWR, &PowerLevel);        
        default:
          return SMPL_BAD_PARAM;
      }
    }



    /*ASCII_to_int*/
    uint8_t ASCII_to_int(uint8_t ascii_vaule)
    {
      switch(ascii_vaule){
        case 0x30:
          return 0;
        case 0x31:
          return 1;                   
        case 0x32:
          return 2;
        case 0x33:
          return 3;
        case 0x34:
          return 4;
        case 0x35:
          return 5;                   
        case 0x36:
          return 6;
        case 0x37:
          return 7;
        case 0x38:
          return 8;
        case 0x39:
          return 9;          
        default:
          return ascii_vaule;
      }
      
    }



    /* itoa:  convert n to characters in s */
    static void itoa(int n,unsigned char s[])
    {
      int i, sign;
 
      if ((sign = n) < 0)  /* record sign */
         n = -n;          /* make n positive */
      i = 0;
      do {       /* generate digits in reverse order */
         s[i++] = n % 10 + '0';   /* get next digit */
      } while ((n /= 10) > 0);     /* delete it */
       if (sign < 0)
         s[i++] = '-';
      s[i] = '\0';
      reverse(s);
    }
    /* itoa:  convert n to characters in s */
    static void itoa_int16(int16_t n,unsigned char s[])
    {
      int16_t i, sign;
 
      if ((sign = n) < 0)  /* record sign */
         n = -n;          /* make n positive */
      i = 0;
      do {       /* generate digits in reverse order */
         s[i++] = n % 10 + '0';   /* get next digit */
      } while ((n /= 10) > 0);     /* delete it */
       if (sign < 0)
         s[i++] = '-';
      s[i] = '\0';
      reverse(s);
    }
     /* reverse:  reverse string s in place */
    static void reverse(unsigned char s[])
    {
      int i, j;
      unsigned char c;
 
      for (i = 0, j = strlen(s)-1; i<j; i++, j--) {
         c = s[i];
         s[i] = s[j];
         s[j] = c;
      }
    }
